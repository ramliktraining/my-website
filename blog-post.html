const urlParams = new URLSearchParams(window.location.search); const slug = urlParams.get('slug'); fetch('posts.json') .then(res => res.json()) .then(posts => { const post = posts.find(p => p.slug === slug); if (post) { document.title = post.title; document.getElementById('post-content').innerHTML = `

${post.title}

Published: ${post.date}

${post.description}

`; } else { document.getElementById('post-content').innerHTML = '

Post not found.

'; } });
